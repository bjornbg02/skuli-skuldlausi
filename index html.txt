<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

canvas.width = 400;
canvas.height = 500;

let player = {x:180,y:440,w:40,h:40};
let coins = [];
let debts = [];

let score = 0;
let lives = 3;

function spawn() {
  coins.push({x:Math.random()*360,y:0});
  if(Math.random()<0.3) debts.push({x:Math.random()*360,y:0});
}

function drawPlayer() {
  ctx.fillStyle="#1f2f5b";
  ctx.fillRect(player.x,player.y,player.w,player.h);
}

function drawCoin(c) {
  ctx.fillStyle="gold";
  ctx.beginPath();
  ctx.arc(c.x,c.y,10,0,Math.PI*2);
  ctx.fill();
}

function drawDebt(d) {
  ctx.fillStyle="red";
  ctx.fillRect(d.x,d.y,20,20);
}

function update() {
  ctx.clearRect(0,0,canvas.width,canvas.height);

  drawPlayer();

  coins.forEach((c,i)=>{
    c.y+=3;
    drawCoin(c);

    if(Math.abs(c.x-player.x)<30 && Math.abs(c.y-player.y)<30){
      score+=1000;
      coins.splice(i,1);
    }
  });

  debts.forEach((d,i)=>{
    d.y+=4;
    drawDebt(d);

    if(Math.abs(d.x-player.x)<30 && Math.abs(d.y-player.y)<30){
      lives--;
      debts.splice(i,1);
    }
  });

  document.getElementById("score").innerText=score;
  document.getElementById("lives").innerText=lives;

  if(lives<=0){
    document.getElementById("overlay").style.display="flex";
    return;
  }

  requestAnimationFrame(update);
}

function move(dir){
  player.x += dir*30;
  player.x = Math.max(0,Math.min(360,player.x));
}

function restart(){
  score=0;
  lives=3;
  coins=[];
  debts=[];
  document.getElementById("overlay").style.display="none";
  update();
}

setInterval(spawn,1000);
update();
</script>
